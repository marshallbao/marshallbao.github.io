```
# show databases;
\l

# 显示数据库的信息
\l databases;

# 查看所有资源
\d

# show tables;
\dt
\dt public.*
\dt test.*


# 详细信息加 +
\dn+
\l+
\d+
# 查看用户和角色
\du

# 查看数据库下的schema
\dn

# 查看表的概要信息
\dt tables;

# desc tables;
\d tables;

# 查看表空间信息
\db+

# 查看表结构
# desc table
\d table

# 查看索引
\di

# 切换数据库
# use database
\c database

# 切换用户
\c - username

# 同时切换
\c dbname username

# sql 查询同 mysql

```
